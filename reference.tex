\begin{equation}
H(f) = \int_{-\infty}^{\infty} \! h(\tau) e^{-j2\pi ft} \, \mathrm{d}\tau
\end{equation}
\begin{equation}
h(t) = \int_{-\infty}^{\infty} \! H(f) e^{j2\pi ft} \, \mathrm{d}f
\end{equation}
